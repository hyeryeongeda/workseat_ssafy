void BubbleSorting(int *value, int size) {
    int x, y, temp;
    for (x = 0; x < size; x++) {
        for (y = 0; y < size - x - 1; y++) {
            if (value[y] < value[y + 1]) { # μ¤λ¦„μ°¨μμ΄λ©΄ value[y] >= value[y + 1]

        }
    }
}
